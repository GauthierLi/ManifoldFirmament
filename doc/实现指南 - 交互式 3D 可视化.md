#### 1.2 数据契约 (`src/models.py`)

**⚠️ 关键修复 - numpy 序列化问题**:

```python
def to_json_dict(self) -> dict:
    """转换为 JSON 可序列化格式"""
    def to_python_type(value):
        """将 numpy 类型转换为 Python 原生类型"""
        if isinstance(value, np.ndarray):
            return value.tolist()
        elif hasattr(value, 'item'):  # numpy 标量类型
            return value.item()
        elif isinstance(value, (list, tuple)):
            return [to_python_type(v) for v in value]
        elif isinstance(value, dict):
            return {k: to_python_type(v) for k, v in value.items()}
        return value
    
    return {
        "points": [
            {
                "x": float(p.x) if hasattr(p.x, 'item') else p.x,
                "y": float(p.y) if hasattr(p.y, 'item') else p.y,
                "z": float(p.z) if hasattr(p.z, 'item') else p.z,
                "image_path": p.image_path,
                "original_index": int(p.original_index),
                "color": [float(c) if hasattr(c, 'item') else c for c in p.color]
            }
            for p in self.points
        ],
        "metadata": to_python_type(self.metadata)
    }
```

**原因**: UMAP 输出的是 `numpy.float32`，Python 的 `json.dump` 无法直接序列化。

#### 2.2 后台处理 + 实时日志

```python
# 全局日志存储
processing_logs = []
processing_status = {"running": False, "completed": False}
log_lock = threading.Lock()

def add_log(message: str):
    """线程安全的日志添加"""
    with log_lock:
        processing_logs.append(message)
        if len(processing_logs) > 1000:  # 限制日志条数
            processing_logs.pop(0)

def run_processing():
    """后台处理函数"""
    with log_lock:
        processing_status["running"] = True
        processing_logs.clear()
    
    try:
        add_log(f"开始处理：{input_file.name}")
        
        process = subprocess.Popen(
            ["python", "src/main.py", str(input_file), "--output", str(output_file)],
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
            text=True
        )
        
        # 实时读取输出
        for line in process.stdout:
            line = line.strip()
            if line:
                add_log(line)  # 发送到前端
        
        process.wait()
        add_log("✅ 处理完成！")
        processing_status["completed"] = True
        
    finally:
        processing_status["running"] = False
```

**⚠️ 关键点**:
- 使用 `subprocess.Popen` + `stdout=PIPE` 实时捕获输出
- 使用 `threading.Thread` 后台运行，不阻塞 API
- 使用 `threading.Lock` 保证线程安全
- 限制日志条数（1000 条）避免内存泄漏

### 3. Three.js 前端渲染 (`src/frontend/viewer.js`)

#### 3.1 点云渲染配置

```javascript
createPointCloud() {
    const geometry = new THREE.BufferGeometry();
    const positions = new Float32Array(points.length * 3);
    const colors = new Float32Array(points.length * 3);
    
    // 填充数据
    points.forEach((p, i) => {
        positions[i * 3] = p.x;
        positions[i * 3 + 1] = p.y;
        positions[i * 3 + 2] = p.z;
        colors[i * 3] = p.color[0];
        colors[i * 3 + 1] = p.color[1];
        colors[i * 3 + 2] = p.color[2];
    });
    
    geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
    geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3));
    
    // ⭐ 关键配置：固定大小的点
    const material = new THREE.PointsMaterial({
        size: 2,                    // 点大小（像素）
        vertexColors: true,         // 使用顶点颜色
        transparent: true,
        opacity: 0.9,
        sizeAttenuation: false      // ⭐ 禁用距离缩放，点大小固定
    });
    
    this.points = new THREE.Points(geometry, material);
    this.scene.add(this.points);
}
```

**⚠️ 关键参数**:
- `size: 2` - 点的大小（像素）
- `sizeAttenuation: false` - **禁用透视缩放**，放大时点大小不变
- `vertexColors: true` - 每个点使用独立颜色

```javascript
makeDraggable(panel) {
    const header = panel.querySelector('.panel-header');
    let isDragging = false;
    let startX, startY, startLeft, startTop;
    
    header.addEventListener('mousedown', (e) => {
        if (e.target.closest('button')) return;  // 忽略按钮点击
        
        isDragging = true;
        startX = e.clientX;
        startY = e.clientY;
        startLeft = panel.offsetLeft;
        startTop = panel.offsetTop;
        
        panel.style.zIndex = 200;  // 提升层级
    });
    
    document.addEventListener('mousemove', (e) => {
        if (!isDragging) return;
        
        const dx = e.clientX - startX;
        const dy = e.clientY - startY;
        
        panel.style.left = (startLeft + dx) + 'px';
        panel.style.top = (startTop + dy) + 'px';
        
        // 显示吸附提示
        this.showSnapIndicator(panel);
    });
    
    document.addEventListener('mouseup', (e) => {
        if (!isDragging) return;
        isDragging = false;
        
        // 应用吸附
        this.applySnap(panel);
        
        // 保存位置
        this.savePanelPosition(panel);
    });
}
```

#### 4.3 边缘吸附

```javascript
applySnap(panel) {
    const rect = panel.getBoundingClientRect();
    const viewportWidth = window.innerWidth;
    const viewportHeight = window.innerHeight;
    const snapDistance = 20;  // 吸附阈值（像素）
    
    let newLeft = rect.left;
    let newTop = rect.top;
    
    // 左右边缘吸附
    if (rect.left < snapDistance) {
        newLeft = 20;  // 吸附到左边
    } else if (rect.right > viewportWidth - snapDistance) {
        newLeft = viewportWidth - rect.width - 20;  // 吸附到右边
    }
    
    // 上下边缘吸附
    if (rect.top < snapDistance) {
        newTop = 20;  // 吸附到顶部
    } else if (rect.bottom > viewportHeight - snapDistance - 60) {
        newTop = viewportHeight - rect.height - 60;  // 吸附到底部
    }
    
    panel.style.left = newLeft + 'px';
    panel.style.top = newTop + 'px';
}
```

#### 4.4 位置保存（localStorage）

```javascript
savePanelPosition(panel) {
    const positions = {};
    this.draggablePanels.forEach(p => {
        positions[p.id] = {
            left: p.offsetLeft,
            top: p.offsetTop
        };
    });
    localStorage.setItem('panelPositions', JSON.stringify(positions));
}

loadPanelPositions() {
    const saved = localStorage.getItem('panelPositions');
    if (!saved) return;
    
    const positions = JSON.parse(saved);
    Object.keys(positions).forEach(id => {
        const panel = document.getElementById(id);
        if (panel && positions[id]) {
            panel.style.left = positions[id].left + 'px';
            panel.style.top = positions[id].top + 'px';
        }
    });
}
```

```javascript
startLogPolling() {
    if (this.logInterval) clearInterval(this.logInterval);
    
    this.logInterval = setInterval(async () => {
        try {
            const response = await fetch('/api/log');
            const result = await response.json();
            
            if (result.logs && result.logs.length > 0) {
                result.logs.forEach(log => {
                    let type = 'info';
                    if (log.includes('✅')) type = 'success';
                    else if (log.includes('❌')) type = 'error';
                    else if (log.includes('步骤')) type = 'highlight';
                    
                    this.addLog(log, type);
                });
            }
            
            if (result.completed) {
                clearInterval(this.logInterval);
                setTimeout(() => this.loadData(), 1500);
            }
        } catch (error) {
            console.error('日志轮询失败:', error);
        }
    }, 1000);  // 每秒轮询一次
}
```

## 🐛 常见问题和解决方案

### 1. JSON 序列化错误

**错误**: `TypeError: Object of type float32 is not JSON serializable`

**解决**: 在 `to_json_dict()` 中转换 numpy 类型：
```python
def to_python_type(value):
    if hasattr(value, 'item'):
        return value.item()
    return value
```

### 2. 点太大/太小

**调整**: 修改 `viewer.js` 中的 `size` 参数：
```javascript
const material = new THREE.PointsMaterial({
    size: 2,  // 调整这个值
});
```

### 3. 点随距离缩放

**解决**: 设置 `sizeAttenuation: false`

### 4. 文件上传 404

**检查**: 
- FastAPI 是否安装 `python-multipart`
- HTML form 的 `enctype` 是否正确

### 5. 日志不更新

**检查**:
- 后台线程是否正常启动
- 前端轮询间隔是否合理
- CORS 配置（如果跨域）
